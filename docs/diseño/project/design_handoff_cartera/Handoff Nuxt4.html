Cartera — Handoff Nuxt 4

App mobile-first de digitalización de tickets y finanzas personales. Este documento es la referencia técnica completa para implementar el prototipo en producción con Nuxt 4.

● Alta Fidelidad Nuxt 4 Vue 3 Composition API Pinia Mobile-first PWA

📋 Overview

Cartera permite a los usuarios fotografiar tickets físicos (con extracción automática por IA), añadirlos manualmente, consultar estadísticas de gasto por periodo/categoría y gestionar su perfil. Diseñada como PWA mobile-first con soporte offline.

Pantallas principales

  • Login / Registro
  • Dashboard con gráficos
  • Lista de tickets
  • Detalle del ticket
  • Escaneo IA / Formulario manual
  • Estadísticas
  • Perfil + subpantallas

Funcionalidades clave

  • OCR + extracción IA (OpenAI Vision)
  • Categorización automática
  • Gráficos día/semana/mes/año
  • Descarga de tickets como imagen
  • Auth con email + OAuth (Google/Apple)
  • PWA con soporte offline

⚡ Composables

useTickets()

CRUD completo de tickets. Fetching con useFetch, caché automática y optimistic updates.

export function useTickets() {
  const store = useTicketsStore()

  const { data: tickets, pending, refresh } = useFetch<Ticket[]>('/api/tickets', {
    default: () => [],
  })

  const createTicket = async (data: CreateTicketDto) => {
    const ticket = await $fetch('/api/tickets', {
      method: 'POST', body: data
    })
    await refresh()
    return ticket
  }

  const deleteTicket = async (id: string) => {
    await $fetch(`/api/tickets/${id}`, { method: 'DELETE' })
    await refresh()
  }

  return { tickets, pending, createTicket, deleteTicket, refresh }
}

useAIExtraction()

Gestiona el flujo de extracción IA: upload de imagen → llamada a OpenAI Vision → parsing de JSON.

export function useAIExtraction() {
  const phase = ref<'idle' | 'uploading' | 'extracting' | 'done' | 'error'>('idle')
  const progress = ref(0)
  const result = ref<ExtractedTicket | null>(null)

  const extract = async (file: File) => {
    phase.value = 'uploading'
    const formData = new FormData()
    formData.append('image', file)

    phase.value = 'extracting'
    // Simula progreso mientras espera la IA
    const interval = setInterval(() => {
      if (progress.value < 90) progress.value += 8
    }, 200)

    try {
      result.value = await $fetch('/api/ai/extract', {
        method: 'POST', body: formData
      })
      progress.value = 100
      phase.value = 'done'
    } catch {
      phase.value = 'error'
    } finally {
      clearInterval(interval)
    }
  }

  return { phase, progress, result, extract }
}

useStats()

Agrega tickets por periodo y categoría para las visualizaciones.

export function useStats(period: Ref<'dia' | 'semana' | 'mes' | 'año'>) {
  const { tickets } = useTickets()

  const filtered = computed(() =>
    tickets.value.filter(t => isInPeriod(t.fecha, period.value))
  )

  const total = computed(() =>
    filtered.value.reduce((s, t) => s + t.total, 0)
  )

  const byCategory = computed(() =>
    groupBy(filtered.value, 'categoria')
  )

  const chartPoints = computed(() =>
    buildTimeSeries(filtered.value, period.value)
  )

  return { total, byCategory, chartPoints, filtered }
}

useDownload()

Genera y descarga el ticket como SVG o PDF desde el cliente.

export function useDownload() {
  const downloadTicketSVG = (ticket: Ticket, el: HTMLElement) => {
    const data = new XMLSerializer().serializeToString(el)
    const blob = new Blob([data], { type: 'image/svg+xml' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `ticket-${ticket.comercio}-${ticket.fecha}.svg`
    a.click()
    URL.revokeObjectURL(url)
  }
  return { downloadTicketSVG }
}

🗄 Pinia Stores

stores/auth.ts

export const useAuthStore = defineStore('auth', () => {
  const { loggedIn, user, clear } = useUserSession()

  const login = async (email: string, password: string) => {
    await $fetch('/api/auth/login', { method: 'POST', body: { email, password } })
  }

  const logout = async () => {
    await $fetch('/api/auth/logout', { method: 'POST' })
    await clear()
    await navigateTo('/login')
  }

  return { loggedIn, user, login, logout }
})

stores/tickets.ts

export const useTicketsStore = defineStore('tickets', () => {
  const tickets = ref<Ticket[]>([])
  const activeFilter = ref<TicketCategoria | null>(null)

  const filtered = computed(() =>
    activeFilter.value
      ? tickets.value.filter(t => t.categoria === activeFilter.value)
      : tickets.value
  )

  return { tickets, activeFilter, filtered }
})

Extracción IA — server handler

// server/api/ai/extract.post.ts
import OpenAI from 'openai'

export default defineEventHandler(async (event) => {
  const formData = await readFormData(event)
  const image = formData.get('image') as File
  const base64 = await fileToBase64(image)

  const openai = new OpenAI({ apiKey: useRuntimeConfig().openaiKey })

  const response = await openai.chat.completions.create({
    model: 'gpt-4o',
    messages: [{
      role: 'user',
      content: [
        { type: 'text', text: `Extrae los datos de este ticket de compra como JSON con:
          { comercio, fecha (ISO), total, iva, items:[{nombre,precio}],
            metodoPago, categoria, confianza (0-1) }` },
        { type: 'image_url', image_url: { url: `data:image/jpeg;base64,${base64}` } }
      ]
    }],
    response_format: { type: 'json_object' },
  })

  return JSON.parse(response.choices[0].message.content!)
})
